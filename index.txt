<script>
/* --- COLOR PALETTE --- */
const PALETTE = [
  { color: '#F41C40', weight: 12 }, // IU crimson hero
  { color: '#990000', weight: 8 },  // strong red
  { color: '#6D0808', weight: 3 },  // depth accent (reduced)
  { color: '#EEEEF0', weight: 3 },  // noticeable highlights
  { color: '#072332', weight: 1 }   // barely present
];

function weightedColor() {
  const total = PALETTE.reduce((s, c) => s + c.weight, 0);
  let r = Math.random() * total;
  for (const c of PALETTE) {
    if ((r -= c.weight) <= 0) return c.color;
  }
}

/* --- SETUP --- */
const background = document.querySelector('.background');
const FIELDS = 5;

let energy = 0;
let mx = window.innerWidth / 2;
let my = window.innerHeight / 2;

function createField(type = 'color') {
  const el = document.createElement('div');

  if (type === 'dark') {
    el.className = 'field dark';
    el.style.background = `
    radial-gradient(
        circle at var(--gx, 50%) var(--gy, 50%),
        #000000,
        #050a10 45%,
        transparent 70%
    )
    `;

  } else {
    const c1 = weightedColor();
    const c2 = weightedColor();

    el.className = 'field';
    el.style.background = `
    radial-gradient(
        circle at var(--gx, 50%) var(--gy, 50%),
        ${c1},
        ${c2} 45%,
        transparent 70%
    )
    `;

  }

  background.appendChild(el);

  return {
    el,
    x: Math.random() * 100,
    y: Math.random() * 100,
    tx: Math.random() * 100,
    ty: Math.random() * 100,
    speed: 0.01 + Math.random() * 0.015
  };
}

const fields = [];

// color fields
for (let i = 0; i < 3; i++) fields.push(createField('color'));

// dark ink fields
for (let i = 0; i < 3; i++) fields.push(createField('dark'));



document.addEventListener('mousemove', e => {
  const dx = e.clientX - mx;
  const dy = e.clientY - my;
  mx = e.clientX;
  my = e.clientY;

  energy += Math.hypot(dx, dy) * 0.002;
  energy = Math.min(energy, 2.5);
});

document.getElementById('scrollCaret').addEventListener('click', () => {
  document.getElementById('mission').scrollIntoView({ behavior: 'smooth' });
});

window.addEventListener('scroll', () => {
  const scrollY = window.scrollY;

  const caret = document.getElementById('scrollCaret');
    caret.style.opacity = scrollY < 50 ? '1' : '0';


  // Show nav after hero
    document.getElementById('nav')
    .classList.toggle('visible', scrollY > window.innerHeight * 0.6);


  // Gradient reacts to scroll depth
  const scrollEnergy = Math.min(scrollY / window.innerHeight, 2);
  energy += scrollEnergy * 0.015;
});

const sections = document.querySelectorAll('.section');

const observer = new IntersectionObserver(
  entries => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        entry.target.classList.add('visible');
      }
    });
  },
  { threshold: 0.2 }
);

sections.forEach(section => observer.observe(section));


function animate() {
  energy *= 0.9;
  const sat = 135 + energy * 25;
    background.style.filter = `blur(140px) saturate(${sat}%)`;

  fields.forEach(f => {
    // very slow autonomous drift
    f.tx += Math.sin(Date.now() * 0.0001) * f.speed;
    f.ty += Math.cos(Date.now() * 0.00013) * f.speed;

    // mouse distorts gradient center, not position
    const dx = (mx / window.innerWidth - 0.5) * energy * 40;
    const dy = (my / window.innerHeight - 0.5) * energy * 40;

    // smooth interpolation (NO velocity)
    f.x += (f.tx - f.x) * 0.02;
    f.y += (f.ty - f.y) * 0.02;

    const gx = 50 + f.x * 0.4 + dx;
    const gy = 50 + f.y * 0.4 + dy;

    f.el.style.setProperty('--gx', `${gx}%`);
    f.el.style.setProperty('--gy', `${gy}%`);

  });

  requestAnimationFrame(animate);
}

let isSnapping = false;

window.addEventListener('wheel', e => {
  if (isSnapping) return;

  const sections = [...document.querySelectorAll('.hero, .section')];
  const current = sections.findIndex(
    s => s.getBoundingClientRect().top >= -10
  );

  if (e.deltaY > 0 && current < sections.length - 1) {
    isSnapping = true;
    sections[current + 1].scrollIntoView({ behavior: 'smooth' });
  } else if (e.deltaY < 0 && current > 0) {
    isSnapping = true;
    sections[current - 1].scrollIntoView({ behavior: 'smooth' });
  }

  setTimeout(() => (isSnapping = false), 900);
});

sections.forEach(section => observer.observe(section));

const navLinks = document.querySelectorAll('.nav a');

const navObserver = new IntersectionObserver(entries => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      navLinks.forEach(link => {
        link.classList.toggle(
          'active',
          link.getAttribute('href').slice(1) === entry.target.id
        );
      });
    }
  });
}, { threshold: 0.6 });

document.querySelectorAll('section').forEach(s => navObserver.observe(s));

const toTop = document.getElementById('toTop');

window.addEventListener('scroll', () => {
  toTop.classList.toggle('visible', window.scrollY > window.innerHeight);
});

toTop.addEventListener('click', () => {
  document.getElementById('hero').scrollIntoView({ behavior: 'smooth' });
});



animate();
</script>
